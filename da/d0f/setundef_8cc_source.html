29 struct SetundefWorker
30 {
31  int next_bit_mode;
32  uint32_t next_bit_state;
33 
34  RTLIL::State next_bit()
35  {
36  if (next_bit_mode == 0)
37  return RTLIL::State::S0;
38 
39  if (next_bit_mode == 1)
40  return RTLIL::State::S1;
41 
42  // xorshift32
43  next_bit_state ^= next_bit_state << 13;
44  next_bit_state ^= next_bit_state >> 17;
45  next_bit_state ^= next_bit_state << 5;
46  log_assert(next_bit_state != 0);
47 
48  return ((next_bit_state >> (next_bit_state & 15)) & 16) ? RTLIL::State::S0 : RTLIL::State::S1;
49  }
50 
51  void operator()(RTLIL::SigSpec &sig)
52  {
53  for (auto &bit : sig)
54  if (bit.wire == NULL && bit.data > RTLIL::State::S1)
55  bit = next_bit();
56  }
57 };
