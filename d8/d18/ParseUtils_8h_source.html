80 static int parseInt(B& in) {
81  int val = 0;
82  bool neg = false;
83  skipWhitespace(in);
84  if (*in == '-') neg = true, ++in;
85  else if (*in == '+') ++in;
86  if (*in < '0' || *in > '9') fprintf(stderr, "PARSE ERROR! Unexpected char: %c\n", *in), exit(3);
87  while (*in >= '0' && *in <= '9')
88  val = val*10 + (*in - '0'),
89  ++in;
90  return neg ? -val : val; }
