48 
49  for (auto &mod_it : design->modules_)
50  {
51  if (!design->selected(mod_it.second))
52  continue;
53 
54  for (auto &c : mod_it.second->cells_)
55  for (auto &p : c.second->connections_)
56  {

58 
59  if (ct.cell_output(c.second->type, p.first)) {
