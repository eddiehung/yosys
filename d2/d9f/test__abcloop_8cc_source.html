29 static uint32_t xorshift32(uint32_t limit) {
30  xorshift32_state ^= xorshift32_state << 13;
31  xorshift32_state ^= xorshift32_state >> 17;
32  xorshift32_state ^= xorshift32_state << 5;
33  return xorshift32_state % limit;
34 }
