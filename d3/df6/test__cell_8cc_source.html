32 static uint32_t xorshift32(uint32_t limit) {
33  xorshift32_state ^= xorshift32_state << 13;
34  xorshift32_state ^= xorshift32_state >> 17;
35  xorshift32_state ^= xorshift32_state << 5;
36  return xorshift32_state % limit;
37 }
